**作用**: 消息队列工具类，支持 Redis 和 RabbitMQ

**核心类**:

##### 1. RedisMQ 类 - Redis Stream 实现
```python
class RedisMQ:
    """基于 Redis Stream 的消息队列"""
    
    def __init__(self, host='localhost', port=6379, db=0):
        self.redis_client = redis.Redis(host, port, db)
    
    def publish(self, stream_name: str, message: Dict) -> bool:
        """发布消息到流"""
        message['timestamp'] = datetime.now().isoformat()
        self.redis_client.xadd(stream_name, message)
        return True
    
    def consume(self, stream_name: str, group_name: str, 
                consumer_name: str, count=1) -> List:
        """消费者组方式读取消息"""
        # 创建消费者组（如果不存在）
        try:
            self.redis_client.xgroup_create(
                stream_name, group_name, id='0', mkstream=True
            )
        except:
            pass  # 组已存在
        
        # 读取消息
        messages = self.redis_client.xreadgroup(
            group=group_name,
            consumer=consumer_name,
            streams={stream_name: '>'},
            count=count
        )
        return messages
    
    def ack(self, stream_name: str, group_name: str, message_id: str):
        """确认消息已处理"""
        self.redis_client.xack(stream_name, group_name, message_id)
```

##### 2. RabbitMQ 类 - RabbitMQ 实现
```python
class RabbitMQ:
    """基于 RabbitMQ 的消息队列"""
    
    def __init__(self, host='localhost', port=5672):
        self.connection = pika.BlockingConnection(
            pika.ConnectionParameters(host=host)
        )
        self.channel = self.connection.channel()
    
    def publish(self, queue_name: str, message: Dict) -> bool:
        """发布消息到队列"""
        self.channel.basic_publish(
            exchange='',
            routing_key=queue_name,
            body=json.dumps(message)
        )
        return True
    
    def consume(self, queue_name: str, callback):
        """消费消息"""
        self.channel.basic_consume(
            queue=queue_name,
            on_message_callback=callback,
            auto_ack=False
        )
        self.channel.start_consuming()
```

##### 3. 工厂函数
```python
def create_mq(mq_type='redis', **kwargs):
    """MQ 工厂函数"""
    if mq_type == 'redis':
        return RedisMQ(**kwargs)
    elif mq_type == 'rabbitmq':
        return RabbitMQ(**kwargs)
```

**使用示例**:
```python

class DataPublisher:
    """数据发布器 - 带重传机制"""
    
    def publish_with_retry(self, data: Dict, sensor_type: str):
        """发布数据，失败时加入重试队列"""
        success = False
        
        # 尝试发布
        try:
            mq_client.publish(Config.RAW_DATA_STREAM, {
                'sensor_type': sensor_type,
                'data': data,
                'timestamp': datetime.now().isoformat()
            })
            success = True
        except Exception as e:
            print(f"发布失败：{e}")
        
        # 失败则加入重试队列
        if not success:
            with retry_lock:
                retry_queue.append({
                    'data': data,
                    'sensor_type': sensor_type,
                    'retry_count': 0,
                    'last_retry': datetime.now()
                })
```

##### 2. 重传线程
```python
def retry_thread_func():
    """定期重试失败的消息"""
    while not stop_event.is_set():
        time.sleep(Config.RETRY_INTERVAL)  # 每 30 秒重试一次
        
        with retry_lock:
            current_time = datetime.now()
            to_remove = []
            
            for item in retry_queue:
                # 检查是否可以重试
                if (current_time - item['last_retry']).seconds >= Config.RETRY_INTERVAL:
                    if item['retry_count'] < Config.MAX_RETRY_COUNT:
                        # 重试
                        try:
                            mq_client.publish(Config.RAW_DATA_STREAM, {
                                'sensor_type': item['sensor_type'],
                                'data': item['data'],
                                'retry_count': item['retry_count'] + 1
                            })
                            to_remove.append(item)
                            print(f"✅ 重传成功：{item['sensor_type']}")
                        except:
                            item['retry_count'] += 1
                            item['last_retry'] = current_time
                    else:
                        # 超过最大重试次数，记录到文件
                        save_failed_data(item)
                        to_remove.append(item)
            
            # 移除已处理的项目
            for item in to_remove:
                retry_queue.remove(item)
```

```python
class ValidationRules:
    """数据验证规则"""
    
    @staticmethod
    def validate_skin_sensor(data: Dict) -> tuple:
        """验证皮肤传感器数据"""
        moisture = data.get('moisture', 0)
        oiliness = data.get('oiliness', 0)
        
        # 水分度：0-100%
        if not (0 <= moisture <= 100):
            return False, f"水分度超出范围：{moisture}"
        
        # 油亮度：0-100%
        if not (0 <= oiliness <= 100):
            return False, f"油亮度超出范围：{oiliness}"
        
        return True, None
    
    @staticmethod
    def validate_environment(data: Dict) -> tuple:
        """验证环境传感器数据"""
        humidity = data.get('humidity', 0)
        temperature = data.get('temperature', 0)
        light_lux = data.get('light_lux', 0)
        
        # 湿度：0-100%
        if not (0 <= humidity <= 100):
            return False, f"湿度超出范围：{humidity}"
        
        # 温度：-10℃ ~ 50℃
        if not (-10 <= temperature <= 50):
            return False, f"温度超出范围：{temperature}"
        
        # 光照度：0-2000Lux
        if not (0 <= light_lux <= 2000):
            return False, f"光照度超出范围：{light_lux}"
        
        return True, None

def main():
    mq = create_mq(Config.MQ_TYPE)
    
    while True:
        messages = mq.consume(
            stream_name=Config.VALIDATED_QUEUE,
            group_name=Config.CONSUMER_GROUP,
            consumer_name=Config.CONSUMER_NAME,
            count=10
        )
        
        for msg_id, msg in messages:
            sensor_type = msg.get('sensor_type')
            data = msg.get('data', {})
            
            # 根据传感器类型验证
            if sensor_type == 'skin':
                valid, error = ValidationRules.validate_skin_sensor(data)
            elif sensor_type == 'environment':
                valid, error = ValidationRules.validate_environment(data)
            
            if valid:
                # 发布到写入队列
                mq.publish(Config.WRITE_QUEUE, msg)
                mq.ack(Config.VALIDATED_QUEUE, 
                      Config.CONSUMER_GROUP, msg_id)
            else:
                # 记录错误日志
                logger.error(f"数据验证失败：{error}")
```

```python
def start_module(module_name: str) -> subprocess.Popen:
    """启动一个模块"""
    process = subprocess.Popen(
        ['python3', module_name],
        stdout=subprocess.PIPE,
        stderr=subprocess.PIPE
    )
    return process

def main():
    modules = [
        'module_receiver.py',
        'module_validator.py',
        'module_writer.py',
        'simulator_mq.py'
    ]

    # 监控进程
    try:
        while True:
```

### 🎯 关键特性

#### 1. 消费者组（Consumer Group）
- **作用**: 支持多个相同功能的模块并行处理
- **实现**: 每个模块使用相同的组名，不同的消费者名
- **效果**: 消息只会被组内的一个消费者处理

#### 2. 消息确认（ACK）
- **作用**: 确保消息不丢失
- **实现**: 处理完成后调用 `xack`
- **效果**: 失败的消息会重新进入待处理队列

```python
# 处理消息
messages = mq.consume(...)
for msg_id, msg in messages:
    # 处理业务逻辑
    process(msg)
    
    # 确认消息
    mq.ack(stream_name, group_name, msg_id)
```

#### 3. 重传机制
- **作用**: 处理失败的消息自动重试
- **实现**: 本地重试队列 + 定期重试线程
- **配置**: 最大 3 次，间隔 30 秒

```python
# 重传逻辑
if publish_failed:
    retry_queue.append({
        'data': data,
        'retry_count': 0
    })

# 定期重试
for item in retry_queue:
    if item['retry_count'] < 3:
        retry_publish(item)
```
