2. APScheduler定时任务详解

什么是APScheduler？为什么要用它？

定时任务的三大应用场景：
• 📊 数据统计：每天凌晨计算前一天的平均值、最大值
• 🧹 数据清理：定期删除过期数据（如保留最近90天）
• 📧 定时通知：每天早上8点发送日报邮件

APScheduler vs Cron的区别：
• ✅ Python原生：直接集成在Flask应用中，无需额外进程
• ✅ 灵活调度：支持interval/cron/date三种触发器
• ✅ 持久化：可将任务存储到数据库，重启后不丢失
• ✅ 动态管理：运行时添加/删除/修改任务

您的项目中的定时任务：
• 任务1：每天凌晨2点，从MongoDB读取昨日原始数据，计算统计值，写入MySQL
• 任务2：每小时检查一次设备在线状态，更新devices表的status字段
• 任务3：每周日凌晨3点，清理90天前的JSON备份文件
# scheduler.py - APScheduler定时任务配置 from apscheduler.schedulers.background import BackgroundScheduler from apscheduler.triggers.cron import CronTrigger from datetime import datetime, timedelta import logging logger = logging.getLogger(__name__) def init_scheduler(app, mongo_client, db): """初始化定时任务调度器""" scheduler = BackgroundScheduler() # 任务1：每日凌晨2点计算统计数据 scheduler.add_job( func=calculate_daily_stats, trigger=CronTrigger(hour=2, minute=0), # 每天2:00执行 id='daily_stats_calculation', name='计算每日统计数据', args=[mongo_client, db], replace_existing=True ) # 任务2：每小时检查设备状态 scheduler.add_job( func=check_device_status, trigger=CronTrigger(minute=0), # 每小时整点执行 id='device_status_check', name='检查设备在线状态', args=[mongo_client, db], replace_existing=True ) # 启动调度器 scheduler.start() logger.info("✅ APScheduler定时任务已启动") return scheduler def calculate_daily_stats(mongo_client, db): """计算前一天的统计数据并写入MySQL""" try: yesterday = (datetime.utcnow() - timedelta(days=1)).date() logger.info(f"开始计算 {yesterday} 的统计数据...") # 1. 从MongoDB读取昨日所有皮肤数据 start_time = datetime.combine(yesterday, datetime.min.time()) end_time = datetime.combine(yesterday, datetime.max.time()) skin_data = mongo_client.find_by_time_range( 'skin_data', start_time, end_time ) # 2. 按设备分组统计 stats_by_device = {} for doc in skin_data: device_id = doc['device_id'] if device_id not in stats_by_device: stats_by_device[device_id] = { 'moisture_sum': 0, 'oiliness_sum': 0, 'temp_max': -999, 'temp_min': 999, 'count': 0 } data = doc['data'] stats = stats_by_device[device_id] stats['moisture_sum'] += data.get('moisture', 0) stats['oiliness_sum'] += data.get('oiliness', 0) stats['temp_max'] = max(stats['temp_max'], data.get('temperature', 0)) stats['temp_min'] = min(stats['temp_min'], data.get('temperature', 999)) stats['count'] += 1 # 3. 写入MySQL for device_id, stats in stats_by_device.items(): if stats['count'] == 0: continue avg_moisture = stats['moisture_sum'] / stats['count'] avg_oiliness = stats['oiliness_sum'] / stats['count'] # 使用upsert：如果存在则更新，不存在则插入 existing = db.session.query(SkinDailyStats).filter_by( device_id=device_id, stat_date=yesterday ).first() if existing: existing.avg_moisture = avg_moisture existing.avg_oiliness = avg_oiliness existing.max_temperature = stats['temp_max'] existing.min_temperature = stats['temp_min'] existing.record_count = stats['count'] else: new_stat = SkinDailyStats( device_id=device_id, stat_date=yesterday, avg_moisture=avg_moisture, avg_oiliness=avg_oiliness, max_temperature=stats['temp_max'], min_temperature=stats['temp_min'], record_count=stats['count'] ) db.session.add(new_stat) db.session.commit() logger.info(f"✅ 完成 {len(stats_by_device)} 个设备的统计") except Exception as e: logger.error(f"❌ 统计任务失败: {str(e)}", exc_info=True) db.session.rollback()

3. Flask-SQLAlchemy CRUD接口

CRUD是什么？
• Create（创建）：POST /api/devices
• Read（读取）：GET /api/devices
• Update（更新）：PUT /api/devices/{id}
• Delete（删除）：DELETE /api/devices/{id}

RESTful API设计规范：
• 使用HTTP方法表示操作类型（GET/POST/PUT/DELETE）
• 使用名词复数作为资源路径（/api/devices而非/api/createDevice）
• 返回统一的JSON格式（{status, message, data}）
• 使用合适的HTTP状态码（200成功/404未找到/500服务器错误）
# routes_devices.py - 设备管理接口 from flask import Blueprint, request, jsonify from models import db, Device device_bp = Blueprint('devices', __name__, url_prefix='/api/devices') @device_bp.route('', methods=['GET']) def list_devices(): """获取所有设备列表""" try: page = request.args.get('page', 1, type=int) per_page = request.args.get('per_page', 20, type=int) pagination = Device.query.paginate(page=page, per_page=per_page, error_out=False) devices = [device.to_dict() for device in pagination.items] return jsonify({ 'status': 'success', 'data': devices, 'total': pagination.total, 'pages': pagination.pages }), 200 except Exception as e: return jsonify({'status': 'error', 'message': str(e)}), 500 @device_bp.route('/<int:device_id>', methods=['GET']) def get_device(device_id): """获取单个设备详情""" device = Device.query.get(device_id) if not device: return jsonify({'status': 'error', 'message': '设备不存在'}), 404 return jsonify({'status': 'success', 'data': device.to_dict()}), 200 @device_bp.route('', methods=['POST']) def create_device(): """创建设备""" try: data = request.get_json() # 验证必填字段 if not data.get('device_id'): return jsonify({'status': 'error', 'message': 'device_id不能为空'}), 400 # 检查是否已存在 if Device.query.filter_by(device_id=data['device_id']).first(): return jsonify({'status': 'error', 'message': '设备ID已存在'}), 409 # 创建新设备 new_device = Device( device_id=data['device_id'], device_name=data.get('device_name', ''), location=data.get('location', ''), status=data.get('status', 'active') ) db.session.add(new_device) db.session.commit() return jsonify({ 'status': 'success', 'message': '设备创建成功', 'data': new_device.to_dict() }), 201 except Exception as e: db.session.rollback() return jsonify({'status': 'error', 'message': str(e)}), 500 @device_bp.route('/<int:device_id>', methods=['PUT']) def update_device(device_id): """更新设备信息""" device = Device.query.get(device_id) if not device: return jsonify({'status': 'error', 'message': '设备不存在'}), 404 try: data = request.get_json() # 更新字段 if 'device_name' in data: device.device_name = data['device_name'] if 'location' in data: device.location = data['location'] if 'status' in data: device.status = data['status'] db.session.commit() return jsonify({ 'status': 'success', 'message': '设备更新成功', 'data': device.to_dict() }), 200 except Exception as e: db.session.rollback() return jsonify({'status': 'error', 'message': str(e)}), 500 @device_bp.route('/<int:device_id>', methods=['DELETE']) def delete_device(device_id): """删除设备""" device = Device.query.get(device_id) if not device: return jsonify({'status': 'error', 'message': '设备不存在'}), 404 try: db.session.delete(device) db.session.commit() return jsonify({ 'status': 'success', 'message': '设备删除成功' }), 200 except Exception as e: db.session.rollback() return jsonify({'status': 'error', 'message': str(e)}), 500
